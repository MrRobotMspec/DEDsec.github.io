// Dark mode toggle const modeToggle = document.querySelector('.mode-toggle'); if (modeToggle) { modeToggle.addEventListener('click', () => { document.body.classList.toggle('dark-mode'); localStorage.setItem('darkMode', document.body.classList.contains('dark-mode')); }); // Check for saved preference if (localStorage.getItem('darkMode') === 'true') { document.body.classList.add('dark-mode'); } }
